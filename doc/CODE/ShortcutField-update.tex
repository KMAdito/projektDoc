private void _update()
{
  boolean empty = getShortcut() == null;
	
  viewerContainer.setVisible(!empty);
  textContainer.setVisible(empty);
  
  crossButton.setVisible(isEditable());
  crossButton.setEnabled(!empty);
}		\end{javacode}
